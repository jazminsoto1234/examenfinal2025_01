# 1. Ejecutar las pruebas con coverage (desde el directorio raíz)
python -m coverage run -m unittest src/test/test_unit.py

# 2. Generar reporte general de cobertura
python -m coverage report

# 3. Generar reporte específico para las clases del modelo
python -m coverage report --include="src\models\*"

# 4. Generar reporte detallado con líneas faltantes
python -m coverage report --show-missing

# 5. Generar reporte HTML
python -m coverage html
